Eljárás EgyszerűCserésRendezés(tömb):
    Ciklus i := 0-tól a (tömb hossza - 2)-ig:
        Ciklus j := (i + 1)-től a (tömb hossza - 1)-ig:
            Ha tömb[i] > tömb[j]:
                // Csere
                temp := tömb[j]
                tömb[j] := tömb[i]
                tömb[i] := temp
        Belső ciklus vége
    Külső ciklus vége
    Visszatérés a rendezett tömbbel

Eljárás MinimumKiválasztásosRendezés(tömb):
    Ciklus i := 0-tól a (tömb hossza - 2)-ig:
        minimum_index := i
        // Legkisebb elem megkeresése a maradékban
        Ciklus j := (i + 1)-től a (tömb hossza - 1)-ig:
            Ha tömb[minimum_index] > tömb[j]:
                minimum_index := j
        Belső ciklus vége
        
        // Csere (a megtalált legkisebbet az i. helyre tesszük)
        seged := tömb[i]
        tömb[i] := tömb[minimum_index]
        tömb[minimum_index] := seged
    Külső ciklus vége
    Visszatérés a rendezett tömbbel
Eljárás vége

Eljárás BinárisKeresés(tömb, keresettÉrték):
    Tömb rendezése (növekvő sorrendbe)
    eleje := 0
    vége := tömb hossza - 1
    
    Ciklus amíg (eleje <= vége):
        közép := (eleje + vége) / 2 (egész osztás)
        
        Ha tömb[közép] egyenlő keresettÉrték:
            Visszatérés közép (megtalált index)
        Különben ha tömb[közép] < keresettÉrték:
            eleje := közép + 1  (jobbra keresünk tovább)
        Különben (ha tömb[közép] > keresettÉrték):
            vége := közép - 1   (balra keresünk tovább)
    Ciklus vége
    
    Visszatérés -1 (nem található)
Eljárás vége
